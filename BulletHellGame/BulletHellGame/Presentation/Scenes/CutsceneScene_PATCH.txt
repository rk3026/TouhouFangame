// In CutsceneScene.cs
// When advancing dialogue, apply per-line typing speed and flash like so:

private void AdvanceDialogue()
{
    currentLineIndex++;
    if (currentLineIndex < _cutsceneData.Dialogue.Count)
    {
        currentDialogue = _cutsceneData.Dialogue[currentLineIndex];
        visibleText = "";
        typeTimer = 0f;
        textDone = false;

        typeDelay = currentDialogue.TypeDelay;

        if (currentDialogue.FlashAfterLine)
        {
            flashActive = true;
            flashTimer = 0.5f;
        }
    }
    else
    {
        EndScene();
    }
}
